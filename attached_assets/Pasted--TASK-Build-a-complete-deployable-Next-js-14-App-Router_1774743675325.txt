  h1: text-4xl md:text-5xl font-bold tracking-tight text-white
  h2: text-2xl font-semibold text-white
  h3: text-lg font-medium text-white
  body: text-sm text-slate-400
  stat: text-3xl md:text-4xl font-bold text-cyan-400

Hover effect (apply to ALL tiles using Framer Motion):
  whileHover={{ scale: 1.015, boxShadow: "0 0 30px rgba(34,211,238,0.12)" }}
  transition={{ type: "spring", stiffness: 300, damping: 20 }}

Card entrance animation (staggered, use motion.div):
  initial={{ opacity: 0, y: 24 }}
  animate={{ opacity: 1, y: 0 }}
  transition={{ delay: index * 0.08, duration: 0.5, ease: "easeOut" }}
